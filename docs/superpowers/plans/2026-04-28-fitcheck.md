# FitCheck Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Chrome MV3 extension that scans secondhand listing pages for garment measurements and compares them to saved body measurements, showing a fit verdict in a popup.

**Architecture:** A site registry maps hostnames to DOM selectors; the content script uses the registry to extract description text and passes it to a shared parser that normalizes and regex-matches measurements into a canonical inch-based set; the popup fetches results via `chrome.tabs.sendMessage`, runs fit logic against the active profile, and renders a verdict grid.

**Tech Stack:** Vanilla JS, Chrome Manifest V3, `chrome.storage.local`, SVG icons, no build step.

// content/parser.js

function parseListingText(str) {
  if (!str || typeof str !== 'string') return {};

  // Pass 1: normalize
  let text = str;
  // Strip HTML entities
  text = text.replace(/&amp;/g, '&').replace(/&nbsp;/g, ' ').replace(/&#?\w+;/g, ' ');
  // Convert cm to inches before lowercasing (units are already latin chars)
  text = text.replace(/(\d+(?:\.\d+)?)\s*cm/gi, (_, n) => {
    const inches = Math.round(parseFloat(n) * 0.3937 * 10) / 10;
    return `${inches}"`;
  });
  // Collapse whitespace, lowercase
  text = text.replace(/\s+/g, ' ').toLowerCase().trim();

  return _extractMeasurements(text);
}

const _SYNONYM_GROUPS = [
  { labels: ['chest', 'across\\s+chest'],                      key: 'chest',     mult: 1 },
  { labels: ['bust'],                                          key: 'chest',     mult: 1 },
  { labels: ['pit\\s*to\\s*pit', 'p2p',
              'armpit\\s*to\\s*armpit'],                        key: 'chest',     mult: 2 },
  { labels: ['waist'],                                         key: 'waist',     mult: 1 },
  { labels: ['hips?'],                                         key: 'hips',      mult: 1 },
  { labels: ['length'],                                        key: 'length',    mult: 1 },
  { labels: ['sleeve'],                                        key: 'sleeve',    mult: 1 },
  { labels: ['inseam'],                                        key: 'inseam',    mult: 1 },
  { labels: ['shoulders?'],                                    key: 'shoulders', mult: 1 },
];

const _UNITS_PATTERN = '(?:"|\'\'|in(?:ch(?:es)?)?)';

function _extractMeasurements(text) {
  const result = {};

  // 1. Shorthand block: B38 W30 H40 (highest priority — unambiguous)
  const block = text.match(/b\s*(\d+(?:\.\d+)?)\s*w\s*(\d+(?:\.\d+)?)\s*h\s*(\d+(?:\.\d+)?)/i);
  if (block) {
    result.chest = parseFloat(block[1]);
    result.waist  = parseFloat(block[2]);
    result.hips   = parseFloat(block[3]);
  }

  // 2 & 3. Label-based patterns per synonym group
  for (const group of _SYNONYM_GROUPS) {
    if (result[group.key] !== undefined) continue;

    const alts = group.labels.join('|');

    // Label-first: "chest: 38"" / "chest 38 inches" / "chest | 38""
    const m1 = text.match(
      new RegExp(`(?:${alts})[:\\s|\\-]+?(\\d+(?:\\.\\d+)?)\\s*${_UNITS_PATTERN}?`, 'i')
    );
    if (m1) {
      const val = Math.round(parseFloat(m1[1]) * group.mult * 10) / 10;
      if (val > 0) { result[group.key] = val; continue; }
    }

    // Number-first fallback: "38" chest" / "30 inch waist"
    const m2 = text.match(
      new RegExp(`(\\d+(?:\\.\\d+)?)\\s*${_UNITS_PATTERN}?\\s*(?:${alts})`, 'i')
    );
    if (m2) {
      const val = Math.round(parseFloat(m2[1]) * group.mult * 10) / 10;
      if (val > 0) result[group.key] = val;
    }
  }

  return result;
}
```

// content/fit.js

const KEY_MEASUREMENTS = ['chest', 'waist', 'hips'];

function compareMeasurement(profileVal, listingVal) {
  const delta = listingVal - profileVal;
  if (Math.abs(delta) <= 1)        return { verdict: 'fit',   label: 'FIT'   };
  if (delta > 1  && delta <= 2)    return { verdict: 'roomy', label: 'ROOMY' };
  if (delta < -1 && delta >= -2)   return { verdict: 'snug',  label: 'SNUG'  };
  if (delta > 2)                   return { verdict: 'big',   label: 'BIG'   };
  return                             { verdict: 'tight', label: 'TIGHT' };
}

function overallVerdict(comparisons) {
  if (comparisons.length < 2) {
    return { verdict: 'check', label: 'Check measurements', note: 'Not enough data to be certain.' };
  }

  const results = comparisons.map(c => ({
    ...c,
    result: compareMeasurement(c.profileVal, c.listingVal)
  }));

  const keyResults = results.filter(r => KEY_MEASUREMENTS.includes(r.key));
  const hasHardFail = keyResults.some(r => r.result.verdict === 'tight' || r.result.verdict === 'big');
  if (hasHardFail) return { verdict: 'wont-fit', label: "Likely won't fit" };

  const greenCount = results.filter(r => r.result.verdict === 'fit').length;
  const keyAmber   = keyResults.some(r => r.result.verdict === 'snug' || r.result.verdict === 'roomy');
  if (greenCount / results.length >= 0.75 && !keyAmber) return { verdict: 'fits', label: 'Likely fits' };

  return { verdict: 'check', label: 'Check measurements' };
}
```

// content/content.js
// Depends on FITCHECK_SITES (sites.js) and parseListingText (parser.js)
// injected first via manifest content_scripts ordering.

chrome.runtime.onMessage.addListener((message, _sender, sendResponse) => {
  if (message.type !== 'scan') return;

  const hostname = window.location.hostname;
  const site     = FITCHECK_SITES[hostname];

  if (!site) {
    sendResponse({ measurements: {}, itemTitle: null, site: null });
    return true;
  }

  // Extract description text — try selectors in order, use first non-empty result
  let descriptionText = '';
  for (const selector of site.descriptionSelectors) {
    const els = document.querySelectorAll(selector);
    if (els.length > 0) {
      const text = Array.from(els).map(el => el.innerText || el.textContent || '').join('\n');
      if (text.trim()) { descriptionText = text; break; }
    }
  }

  // Extract item title
  let itemTitle = null;
  for (const sel of site.titleSelector.split(',').map(s => s.trim())) {
    const el = document.querySelector(sel);
    if (el && el.textContent.trim()) { itemTitle = el.textContent.trim(); break; }
  }

  sendResponse({
    measurements: parseListingText(descriptionText),
    itemTitle,
    site: hostname.replace('www.', ''),
  });

  return true;
});
```

    // ── Storage helpers ───────────────────────────────────────────────────────
    function getStorage() {
      return new Promise(resolve =>

    }

    function setStorage(data) {
      return new Promise(resolve => chrome.storage.local.set(data, resolve));

    // ── Profile dropdown ──────────────────────────────────────────────────────
    async function initProfileDropdown() {
      const { profiles = [], activeProfileId } = await getStorage();
      const select = document.getElementById('profile-select');
      select.innerHTML = '';

      if (profiles.length === 0) {
        const opt = document.createElement('option');
        opt.value = '';
        opt.textContent = 'No profiles';
        select.appendChild(opt);
        return null;
      }

      profiles.forEach(p => {
        const opt = document.createElement('option');
        opt.value = p.id;
        opt.textContent = p.name;
        if (p.id === activeProfileId) opt.selected = true;
        select.appendChild(opt);
      });

      select.onchange = async () => {
        await setStorage({ activeProfileId: select.value });
        if (_lastScanResult) renderMain(_lastScanResult, select.value);
      };

      return activeProfileId || profiles[0]?.id || null;
    }

    // ── Settings button ───────────────────────────────────────────────────────
    document.getElementById('btn-settings').addEventListener('click', () => {
      _editingProfileId = null; // always open settings showing the active profile
      showView('settings');
      renderSettings();
    });

    document.getElementById('btn-back').addEventListener('click', () => {
      showView(_lastView);
    });

    // ── Startup ───────────────────────────────────────────────────────────────
    async function startup() {
      showView('loading');

      const activeProfileId = await initProfileDropdown();
      const { profiles = [] } = await getStorage();

      if (profiles.length === 0) {
        showView('settings');
        renderSettings();
        return;
      }

      let responded = false;

      const timer = setTimeout(() => {
        if (responded) return;
        responded = true;
        _lastView = 'empty';
        showView('empty');
        initEmptyView();
      }, 2000);

      chrome.tabs.query({ active: true, currentWindow: true }, tabs => {
        if (!tabs[0]) {
          clearTimeout(timer);
          if (!responded) { responded = true; _lastView = 'empty'; showView('empty'); initEmptyView(); }
          return;
        }

        chrome.tabs.sendMessage(tabs[0].id, { type: 'scan' }, response => {
          if (chrome.runtime.lastError || !response) {
            clearTimeout(timer);
            if (!responded) { responded = true; _lastView = 'empty'; showView('empty'); initEmptyView(); }
            return;
          }

          clearTimeout(timer);
          if (responded) return;
          responded = true;
          _lastScanResult = response;

          const hasMeasurements = Object.keys(response.measurements || {}).length > 0;
          if (hasMeasurements) {
            renderMain(response, activeProfileId);
          } else {
            _lastView = 'empty';
            showView('empty');
            initEmptyView();
          }
        });
      });

    // ── Main view ─────────────────────────────────────────────────────────────
    async function renderMain(scanResult, activeProfileId) {
      const { profiles = [] } = await getStorage();
      const profile = profiles.find(p => p.id === activeProfileId);

      if (!profile) {
        _lastView = 'empty'; showView('empty'); initEmptyView(); return;
      }

      const listing = scanResult.measurements || {};
      const body    = profile.measurements || {};

      const comparisons = CANONICAL
        .filter(key => listing[key] !== undefined && body[key] !== undefined)
        .map(key => ({ key, profileVal: body[key], listingVal: listing[key] }));

      if (comparisons.length === 0) {
        _lastView = 'empty'; showView('empty'); initEmptyView(); return;
      }

      // Verdict banner
      const verdict = overallVerdict(comparisons);
      const banner  = document.getElementById('verdict-banner');
      banner.className = `verdict-banner verdict-${verdict.verdict}`;

      const icons = { fits: '✓', 'wont-fit': '✗', check: '?' };
      document.getElementById('verdict-icon').textContent = icons[verdict.verdict] ?? '?';
      document.getElementById('verdict-text').textContent = verdict.label;
      const passCount = comparisons.filter(c => compareMeasurement(c.profileVal, c.listingVal).verdict === 'fit').length;
      document.getElementById('verdict-sub').textContent =
        verdict.note || `${passCount} of ${comparisons.length} measurements pass`;

      // Measurement grid
      const grid = document.getElementById('measurement-grid');
      grid.innerHTML = '';
      comparisons.forEach(c => {
        const r   = compareMeasurement(c.profileVal, c.listingVal);
        const row = document.createElement('div');
        row.className = 'measurement-row';
        row.innerHTML = `
          <span class="m-name">${c.key}</span>
          <span class="m-values">you ${c.profileVal}" vs <strong>${c.listingVal}"</strong></span>
          <span class="badge badge-${r.verdict}">${r.label}</span>
        `;
        grid.appendChild(row);
      });

      // Footer
      const parts = [scanResult.site, scanResult.itemTitle ? `"${scanResult.itemTitle}"` : null]
        .filter(Boolean);
      document.getElementById('listing-footer').textContent = parts.join(' · ');

      _lastView = 'main';
      showView('main');
    }

      // Measurement editor for selected profile
      const selected     = profiles.find(p => p.id === selectedId);
      const measureLabel = document.getElementById('measure-section-label');
      const editList     = document.getElementById('measure-edit-list');
      const addRow       = document.getElementById('add-measure-row');
      const addForm      = document.getElementById('add-measure-form');

      if (!selected) {
        measureLabel.style.display = 'none';
        editList.innerHTML = '';
        addRow.style.display = 'none';
        addForm.classList.remove('visible');
        return;
      }

      measureLabel.textContent = `${selected.name}'s measurements`;
      measureLabel.style.display = 'block';
      addRow.style.display = 'block';
      editList.innerHTML = '';

      const measurements = selected.measurements || {};
      Object.entries(measurements).forEach(([key, val]) => {
        const row = document.createElement('div');
        row.className = 'measure-edit-row';
        row.innerHTML = `
          <label>${key}</label>
          <input type="number" min="1" max="100" step="0.5" value="${val}" data-key="${key}">
          <span class="unit">"</span>
        `;
        row.querySelector('input').addEventListener('change', async e => {
          const v = parseFloat(e.target.value);
          if (isNaN(v) || v <= 0) return;
          selected.measurements[key] = v;
          await setStorage({ profiles, activeProfileId });
        });
        editList.appendChild(row);
      });

      // Add-measurement inline form
      addRow.onclick = () => {
        addForm.classList.add('visible');
        // Hide keys already set
        Array.from(document.getElementById('new-key').options).forEach(opt => {
          opt.hidden = measurements[opt.value] !== undefined;
        });
      };

      document.getElementById('btn-save-measure').onclick = async () => {
        const key = document.getElementById('new-key').value;
        const val = parseFloat(document.getElementById('new-val').value);
        if (isNaN(val) || val <= 0) return;
        selected.measurements       = selected.measurements || {};
        selected.measurements[key]  = val;
        await setStorage({ profiles, activeProfileId });
        addForm.classList.remove('visible');
        document.getElementById('new-val').value = '';
        renderSettings();
      };
    }

    // New profile button
    document.getElementById('btn-new-profile').addEventListener('click', async () => {
      const name = prompt('Profile name:');
      if (!name?.trim()) return;
      const id = name.trim().toLowerCase().replace(/\s+/g, '-').replace(/[^a-z0-9-]/g, '');
      const { profiles = [], activeProfileId } = await getStorage();
      if (profiles.find(p => p.id === id)) { alert('A profile with that name already exists.'); return; }
      profiles.push({ id, name: name.trim(), measurements: {} });
      await setStorage({ profiles, activeProfileId: activeProfileId || id });
      _editingProfileId = id;
      renderSettings();
      initProfileDropdown();
    });
```
